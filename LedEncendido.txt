;Encender LED

#include <xc.inc>

;bits de configuración 
CONFIG FOSC = INTOSCIO_EC  ; Oscilador interno
CONFIG WDT = OFF    ;Watchdog Timer deshabilitado
CONFIG LVP = OFF    ;Low voltage programming deshabilitado
CONFIG PBADEN=OFF

; Vector reset
PSECT resetVec, class=CODE, reloc=2
  GOTO   main

; Código principal
PSECT main_code, class=CODE, reloc=2
main:    
    MOVLW  0     ; Carga el valor 0 (00000000 en binario) en el registro W
    MOVWF  TRISB ;mueve ese valor de W al registro TRISB
    MOVLW  1     ; Cargar el valor 1 (00000001 en binario) en el registro W
    MOVWF  LATB  ;mueve el valor de w al registro LASTB

    END
;fin
